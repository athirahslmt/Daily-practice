CREATE TABLE Department(
	id     INT PRIMARY KEY,
  	name   VARCHAR(50)
);

CREATE TABLE Employees(
	id     INT PRIMARY KEY,
  	name   VARCHAR(50),
  	department_id INT,
  	salary INT,
  	city VARCHAR(50),
  	hiring_year INT,
  	FOREIGN KEY (department_id) REFERENCES departments(id)
);
